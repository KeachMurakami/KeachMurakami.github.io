```r
library(dplyr)
library(magrittr)
library(lubridate)
library(data.table)

Bus_route_files <-
  dir("~/Dropbox/Symp/2016AusPheno/Canberra_Bus/", pattern = "csv", full.names = T)

Timetable <-
  lapply(Bus_route_files, function(files){
    timetable <-
      files %>%
      fread
    
    forSort <-
      timetable %>%
      select(2) %>%
      set_names("sort")
              
    timetable %>%
      bind_cols(., forSort) %>%
      na.omit %>%
      mutate(sort = hm(sort)) %>%
      arrange(sort) %>%
      select(-sort)
  })
```
